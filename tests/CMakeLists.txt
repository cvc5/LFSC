macro(lfsc_test file)
  add_test(
    NAME ${file}
    COMMAND ${test_script} $<TARGET_FILE:lfscc> ${CMAKE_CURRENT_LIST_DIR}/tests/${file}
    WORKING_DIRECTORY ${CMAKE_CURRENT_LIST_DIR}
  )
  set_tests_properties(${file} PROPERTIES TIMEOUT 40)
endmacro()

  lfsc_test(${file})
endforeach()
